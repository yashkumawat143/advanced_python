Create a Student class with:
name
marks (list)
Methods:
Calculate average marks
display the student detail



answer :


new=[];
class student:
    
    def __init__(self,name,marks):
        self.name=name;
        self.marks=marks;
    def show(self):
        print("The marks of student  ");
        for mark in self.marks:
            new.append(mark);
    def average(self):
        sum=0;
        print("The average ");
        for mark in self.marks:
            sum=sum+mark;
        return sum/len(self.marks);
    def detail(self):
        print("student name = ",self.name);
        print("Student marks = ",self.marks);
        print(self.average());

Create a Person class.

answer :


class person:
    def __init__(self,name,age):
        self.name=name;
        self.age=age;
    def detail(self):
        print("The name : ",self.name);
        print("The age : ",self.age);

class shape:
    def __init__(self):
        print("The shapes");

class circle(shape):
    def __init__(self,radius):
        self.radius=radius;
    def detail(self):
        print("Area of circle : ",3.14*self.radius*self.radius);

class square(shape):
    def __init__(self,side):
        self.side=side;
    def detail2(self):
        print("Square : ",self.side*self.side);

9......Payment System

Create:
Payment

Each class should have pay() method.



answer :

class payment:
    def __init__(self,amount):
        self.amount=amount;

class creditcard(payment):
    def __init__(self,amount,pin):
        super().__init__(amount);
        self.pin=pin;
    def pay(self):
        print("Payment successfull",self.amount);

class upi(payment):
    def __init__(self,amount,pin):
        super().__init__(amount);
        self.pin=pin;
    def pay(self):
        print("Payment successfull",self.amount);

s=creditcard(5000,123);
s.pay();
