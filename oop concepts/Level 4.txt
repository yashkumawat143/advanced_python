Make balance private.
Provide:
deposit()
withdraw()
get_balance()

answer :

class privatebalance:
    def __init__(self,amount):
        self.amount=amount;
    def deposite(self,dep):
        self.amount+=dep;
        print("Amount successfully added",dep);
    def withdraw(self,wi):
        if(self.amount>=wi):
            self.amount-=wi;
            print("Successfully withdrawl",wi);
        else:
            print("no sufficient balance");
    def getbalance(self):
        print("The balance = ",self.amount);

2......employee Salary

Create private salary attribute.
Use getter and setter methods.

answer :


class employee:
    def __init__(self,salary):
        self.__salary=salary;
    
    def getter(self):
        return self.__salary;
   
    def setter(self,salary):
        self.__salary=salary;

s=employee(600000);
print("The salary ",s.getter());

s.setter(700000);
print("The updated salary ",s.getter());

4. Add Two Complex Numbers

Create Complex class.
Overload + operator.

answer :


class complex:
    def __init__(self,real,img):
        self.real=real;
        self.img=img;
    def __add__(self,other):
        real=self.real+other.real;
        img=self.img+other.img;
        return complex(real,img);
    def display(self):
        print(f"{self.real}+{self.img}i")
c=complex(3,4);
c1=complex(5,6);

cc=c+c1;
cc.display()
