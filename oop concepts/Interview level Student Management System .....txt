class student:
    def __init__(self,name,rollno,marks):
        self.name=name;
        self.rollno=rollno;
        self.marks=marks;
    def detail(self):
        print(f"name : {self.name}");
        print(f"roll : {self.rollno}");
        print(f"marks : {self.marks}");

class studentmanagement:
    def __init__(self):
        self.student=[];
    def add_student(self):
        name=input("Enter the name: ");
        rollno=int(input("Enter the roll no : "));
        marks=int(input("Enter the marks : "));
        a=student(name,rollno,marks);
        self.student.append(a);

        print("Student added successfully");
    
    def search_stu(self):
        n1=int(input("Enter the roll no : "));
        for i in self.student:
            if(i.rollno==n1):
                print("student found");
                i.detail();
                return;
        
        print("Student not found");
 
    def dele(self):
        n1=int(input("Enter the roll number do you want to delete : "));
        for i in self.student:
            if(i.rollno==n1):
                self.student.remove(i);
                print("Deletd successfully");
                return;
            
        print("student not found");
 
    def dif(self):
        for i in self.student:
            i.detail();
