Q....print the name and marks of three subject and also find average.

answer :


class student:
    def __init__(self,name,marks):
        self.name=name;
        self.marks=marks;
    def average(self):
        sum=0
        for i in self.marks:
            sum=sum+i;
        print(self.name,"score is",sum/3);
ans=student("Yash",[12,34,45]);
ans.average();
